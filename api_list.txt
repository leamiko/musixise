
—— MUSIXISER 音乐人相关接口 ——

#音乐人注册 P0
musixiser.register
	*nickname,  *password, realname, tel, email, birth, gender, smallAvatar, largeAvatar,nation
	token
	
#音乐人加v(后台) P1
musixiser.masterize(大师化)
	isMaster(默认为0，大师为1)

#音乐人获取个人信息 P2
#musixiser.getInfo
	token
	nickname, realname, tel, email, birth, gender, smallAvatar, largeAvatar,nation
	
#音乐人更新个人信息 P2
musixiser.updateInfo
	*token, nickname,  password, realname, tel, email, birth, gender, smallAvatar, largeAvatar,nation

#音乐人更新密码 P1
#musixiser.changePassword
	token, oldPassword, newPassword

#音乐人登陆 P0
musixiser.login
	nickname/tel/email, password
	token

#音乐人开始演出 P0
musixiser.onStage
	token
	workId

#音乐人中止演出 P0
musixiser.offStage
	token,workId

#音乐人登出 P0
musixiser.logout
	token

#储存音乐人表演的作品信息 P2
musixiser.saveWork
	token, MidiStr,workId

#获取音乐人过去的演出作品列表（midi或其他形式返回）P2
musixiser.pastWorksList
	token, page, limit
	return list: mp3/midifile
